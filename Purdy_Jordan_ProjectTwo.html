def play_game(model, qmaze, pirate_cell, max_steps=None):
    qmaze.reset(pirate_cell)
    envstate = qmaze.observe()
    steps = 0
    if max_steps is None:
        max_steps = qmaze.maze.size * 4  # safety cutoff

    while steps < max_steps:
        # Check valid actions
        valid_actions = qmaze.valid_actions()
        if not valid_actions:
            return False # Trapped
            
        state = np.asarray(envstate, dtype=np.float32)
        if state.ndim == 1:
            state = np.expand_dims(state, axis=0)

        q_values = model(state, training=False).numpy()[0]
        
        #Force the agent to pick the highest Q-value out of only the valid actions
        action = max(valid_actions, key=lambda a: q_values[a])

        envstate, reward, game_status = qmaze.act(action)
        steps += 1

        if game_status == 'win':
            return True
        elif game_status == 'lose':
            return False

    return False  # timed out with no result

def completion_check(model, maze_or_qmaze, max_steps=None):
    # Accept either raw numpy maze or TreasureMaze instance
    if isinstance(maze_or_qmaze, TreasureMaze):
        qmaze = maze_or_qmaze
    else:
        qmaze = TreasureMaze(maze_or_qmaze)

    for cell in qmaze.free_cells:
        if not qmaze.valid_actions(cell):
            continue
        if not play_game(model, qmaze, cell, max_steps=max_steps):
            print("Failed starting cell:", cell)
            return False
    return True

def qtrain(model, maze, **opt):
    #Train the pirate in 3 stages
    #1. Generate a successful path with breadth first search (BFS)
    #2. Warm up the neural network on those paths
    #3. fine tune the policy with Deep Q
    
    global epsilon

    loss_history = []
    win_rate_history = []
    epsilon_history = []
    reward_history = []

    n_epoch = opt.get("n_epoch", 2000)
    max_memory = opt.get("max_memory", 5000)
    data_size = opt.get("data_size", 64)

    start_time = datetime.datetime.now()
    qmaze = TreasureMaze(maze)

    target_model = tf.keras.models.clone_model(model)
    target_model.set_weights(model.get_weights())

    experience = GameExperience(model, target_model, max_memory=max_memory)

    hsize = 100
    win_history = []

    epsilon = opt.get("epsilon", 0.10)
    epsilon_decay = opt.get("epsilon_decay", 0.995)
    epsilon_min = opt.get("epsilon_min", 0.02)

    #This is a BFS helper
    #Purpose is to generate guaranteed valid paths to the treasure,
    #as pure random exploration was not reliable at all in this maze
    def shortest_path_actions(start):
        target = (maze.shape[0] - 1, maze.shape[1] - 1)

        moves = {
            LEFT:  (0, -1),
            UP:    (-1, 0),
            RIGHT: (0, 1),
            DOWN:  (1, 0)
        }

        queue = [(start, [])]
        visited = {start}

        while queue:
            cell, path = queue.pop(0)

            if cell == target:
                return path

            r, c = cell

            for action, (dr, dc) in moves.items():
                nr, nc = r + dr, c + dc
                nxt = (nr, nc)

                if nr < 0 or nr >= maze.shape[0]:
                    continue
                if nc < 0 or nc >= maze.shape[1]:
                    continue
                if maze[nr, nc] == 0.0:
                    continue
                if nxt in visited:
                    continue

                visited.add(nxt)
                queue.append((nxt, path + [action]))

        return None

    #This is the warm up/warm starting data
    #Builds surpervised training examples from the successful paths
    #Learns what good actions look like before training starts and reinforcement begins
    print("Building warm-start training data...")

    X = []
    y = []

    all_start_cells = list(qmaze.free_cells)

    for start_cell in all_start_cells:
        path = shortest_path_actions(start_cell)

        if path is None:
            continue

        qmaze.reset(start_cell)
        envstate = qmaze.observe()

        for action in path:
            #corrrect action from shortest path is given highest target value,
            #while other actions are pushed lower to teach perferred move
            target_q = np.zeros(num_actions, dtype=np.float32)
            target_q[:] = -1.0
            target_q[action] = 1.0

            X.append(envstate[0])
            y.append(target_q)

            prev_envstate = envstate
            envstate, reward, game_status = qmaze.act(action)

            experience.remember([
                prev_envstate,
                action,
                reward,
                envstate,
                game_status != "not_over"
            ])

            if game_status == "win":
                break

    X = np.array(X, dtype=np.float32)
    y = np.array(y, dtype=np.float32)

    print("Warm-start samples:", len(X))

    #Pretraining like this gives a strong initial policy instead of demanding it
    #discovers the maze through trial and error, which was not working
    print("Training model on known successful paths...")

    for i in range(300):
        h = model.fit(X, y, epochs=1, batch_size=32, verbose=0)

        if i % 50 == 0:
            print(f"Warm-start epoch {i}/300 | Loss={h.history['loss'][0]:.4f}")

    target_model.set_weights(model.get_weights())

    print("Checking model after warm-start...")
    warm_start_passed = completion_check(model, qmaze)
    
    if warm_start_passed:
        print("Warm-start produced a working initial policy.")
    else:
        print("Warm-start did not fully solve the maze, continuing with DQN fine-tuning...")
    
    print("Continuing with DQN fine-tuning to validate and stabilize the learned policy...")

    #Now begins the reinforcement training and Deep Q
    dqn_start_index = len(win_history)
    for epoch in range(n_epoch):

        if epoch % 4 == 0:
            start_cell = (0, 0)
        else:
            start_cell = random.choice(qmaze.free_cells)

        qmaze.reset(start_cell)
        envstate = qmaze.observe()

        game_over = False
        steps = 0
        max_steps = maze.size * 4
        loss = 0

        #Below is epsilon greedy, which balances exploration and exploitation,
        #Random actions can help discover new paths, while greedy actions used learned Q values
        while not game_over and steps < max_steps:
            valid_actions = qmaze.valid_actions()

            if np.random.rand() < epsilon:
                action = random.choice(valid_actions)
            else:
                q = model.predict(envstate, verbose=0)[0]
                action = max(valid_actions, key=lambda a: q[a])

            prev_envstate = envstate
            envstate, reward, game_status = qmaze.act(action)
            
            #remember actions from batch of past experiences
            experience.remember([
                prev_envstate,
                action,
                reward,
                envstate,
                game_status != "not_over"
            ])

            steps += 1

            if game_status == "win":
                win_history.append(1)
                game_over = True

            elif game_status == "lose":
                win_history.append(0)
                game_over = True

        if not game_over:
            win_history.append(0)
        
        if len(experience.memory) >= data_size:
            inputs, targets = experience.get_data(data_size)
            h = model.fit(inputs, targets, epochs=1, verbose=0)
            loss = h.history["loss"][0]
            
        #update target network to stabilize q learning targets and reduce oscillation
        if epoch % 25 == 0:
            target_model.set_weights(model.get_weights())

        #moving win rate window allows me to measure recent performance instead of averaging
        #over the entire history, uses however many epochs so far, to allow smooth curve
        window = min(len(win_history), hsize)
        win_rate = sum(win_history[-window:]) / window if window > 0 else 0

        epsilon = max(epsilon * epsilon_decay, epsilon_min)

        dt = datetime.datetime.now() - start_time
        t = format_time(dt.total_seconds())

        print(
            "Epoch: {:03d}/{:d} | Loss: {:.4f} | Episodes: {:d} | Win count: {:d} | Win rate: {:.3f} | Epsilon: {:.3f} | time: {}".format(
                epoch,
                n_epoch - 1,
                loss,
                epoch + 1,
                sum(win_history),
                win_rate,
                epsilon,
                t
            )
        )

        loss_history.append(loss)
        win_rate_history.append(win_rate)
        epsilon_history.append(epsilon)
        reward_history.append(qmaze.total_reward)
        
        #finally, completion check, verifies the policy can solve the maze from every
        #starting position, then ends model
        if epoch % 100 == 0 and epoch >= 300:
            print("Checking completion after DQN fine-tuning...")
            if completion_check(model, qmaze):
                print("Success: Maze fully solved after DQN fine-tuning at epoch", epoch)
                break

    total_time = format_time((datetime.datetime.now() - start_time).total_seconds())

    #for graphing purposes
    dqn_win_history = win_history[dqn_start_index:]
    
    dqn_win_rate_history = []
    for i in range(len(dqn_win_history)):
        start = max(0, i - hsize + 1)
        window = dqn_win_history[start:i+1]
        dqn_win_rate_history.append(sum(window) / len(window))


    #Below are graphs to analyze progress throughout policy journey
    print("Training complete in:", total_time)
    print("\nTraining Summary")
    print("------------------")
    print(f"Epochs completed: {epoch + 1}")
    print(f"Final win rate: {win_rate:.3f}")
    print(f"Final epsilon: {epsilon:.3f}")
    print(f"Total wins: {sum(win_history)}")
    print(f"Training time: {total_time}")

    #Win rate once DQN started
    plt.figure(figsize=(10,5))
    plt.plot(dqn_win_rate_history)
    plt.title("DQN Fine-Tuning Win Rate")
    plt.xlabel("DQN Epoch")
    plt.ylabel("Win Rate")
    plt.ylim(0, 1.05)
    plt.show()

    #training loss
    plt.figure(figsize=(10,5))
    plt.plot(loss_history)
    plt.title("Training Loss")
    plt.xlabel("Epoch")
    plt.ylabel("Loss")
    plt.show()

    #epsilon/exploration
    plt.figure(figsize=(10,5))
    plt.plot(epsilon_history)
    plt.title("Exploration Rate")
    plt.xlabel("Epoch")
    plt.ylabel("Epsilon")
    plt.show()

    #success per episode
    plt.figure(figsize=(10,5))
    plt.plot(dqn_win_history)
    plt.title("DQN Episode Outcomes")
    plt.xlabel("DQN Epoch")
    plt.ylabel("Outcome (1 = win, 0 = loss)")
    plt.ylim(-0.05, 1.05)
    plt.show()

    return model

def format_time(seconds):
    if seconds < 400:
        s = float(seconds)
        return "%.1f seconds" % (s,)
    elif seconds < 4000:
        m = seconds / 60.0
        return "%.2f minutes" % (m,)
    else:
        h = seconds / 3600.0
        return "%.2f hours" % (h,)
